## Pipeline Responses

### AssistantPipelineResponse

```typescript
interface AssistantPipelineResponse {
  mode: "assistant" | "dictation";
  selectionRewrite: boolean;
  selectionPending: boolean;
  selectionContextCleared: boolean;
  selectionContextUsed: boolean;
  transcript: string;
  assistantResponse: string;
  audioBase64: string;
  sttLatencyMs: number;
  aiLatencyMs: number;
  ttsLatencyMs: number;
  totalLatencyMs: number;
}
```

Complete response from the assistant processing pipeline.

<ResponseField name="mode" type="string">
  Pipeline mode: `"assistant"` (AI response) or `"dictation"` (transcription only)
</ResponseField>

<ResponseField name="selectionRewrite" type="boolean">
  Whether the request was to rewrite selected text
</ResponseField>

<ResponseField name="selectionPending" type="boolean">
  Selection popup is awaiting user action
</ResponseField>

<ResponseField name="selectionContextCleared" type="boolean">
  Previously stored selection context was cleared
</ResponseField>

<ResponseField name="selectionContextUsed" type="boolean">
  Selection context was used in the request
</ResponseField>

<ResponseField name="transcript" type="string">
  Transcribed text from speech input
</ResponseField>

<ResponseField name="assistantResponse" type="string">
  AI-generated response text
</ResponseField>

<ResponseField name="audioBase64" type="string">
  Base64-encoded TTS audio output
</ResponseField>

<ResponseField name="sttLatencyMs" type="number">
  Speech-to-text processing time in milliseconds
</ResponseField>

<ResponseField name="aiLatencyMs" type="number">
  AI model inference time in milliseconds
</ResponseField>

<ResponseField name="ttsLatencyMs" type="number">
  Text-to-speech generation time in milliseconds
</ResponseField>

<ResponseField name="totalLatencyMs" type="number">
  Total pipeline latency in milliseconds
</ResponseField>
